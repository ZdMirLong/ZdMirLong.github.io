identity server 4

  1. 使用vscode创建项目

    首先vscode安装c# nugget扩展
    cd到需要创建项目的目录 执行命令
    dotnet new webapi –name IdentityCenter

  2. 添加identityserver4引用

    vscode下按住 ctrl+p 选中nugget add package ,输入 identityserver,选择安装 ,安装后点击restore

cd到创建的项目目录下,重新加载项目,否则命名空间无法出现智能提示 cd .\IdentityCenter\ 执行命令 dotnet restore

    首先starup中ConfigureServices方法启用中间件

在Configure方法中启用中间件

 public static IWebHost BuildWebHost(string[] args) =>
        WebHost.CreateDefaultBuilder(args)
            .UseStartup<Startup>()
            .UseUrls("http://localhost:5000")//设置启动端口为5000
            .Build();
}

using System.Collections;
using System.Collections.Generic;
using IdentityServer4.Models;

namespace IdentityCenter
{

public class config
{

    ///获取所有的api集合
    public static IEnumerable<ApiResource> GetResource()
    {
        return new List<ApiResource>{
        new ApiResource("api","My Api")
    };


    }

    /// <summary>
    /// 获取所有客户端
    /// </summary>
    /// <returns></returns>
    public static IEnumerable<Client> GetClients()
    {
        return new List<Client>{

     new Client{
           ClientId="client",//当前客户端id
           AllowedGrantTypes=GrantTypes.ClientCredentials,//使用客服端方式
           ClientSecrets={
              new Secret("Secret".Sha256())
            },
            AllowedScopes={"api"}//指定可以访问的api 与上个方法定义的一致

     }
 };

    }

}

}

public void ConfigureServices(IServiceCollection services)
{
//添加identityserver中间件
services.AddIdentityServer()
.AddDeveloperSigningCredential()
.AddInMemoryApiResources(config.GetResource())//添加config.cs中定义的客户端和服务端
.AddInMemoryClients(config.GetClients());
//添加mvc中间件
services.AddMvc();
}
`
