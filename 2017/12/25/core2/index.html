 /// <summary>
/// 接受数据库连接字符窜
/// </summary>
public class ConnectionString {
    /// <summary>
    /// sql数据库 连接字符串
    /// </summary>
    public string SqlConnectionString { get; set; }
    /// <summary>
    /// mongodb 数据库
    /// </summary>
    public string MongoDbString { get; set; }
}

public class ApplicationConfiguration
{

    public string FileUpPath { get; set; }
   public bool IsSingleLogin{ get; set; }
    public string AttachExtension { get; set; }
    public int AttachImagesize { get; set; }
}

///读取配置文件字符窜
services.Configure(this.Configuration.GetSection(“ConnectionString”));

//读取配置文件
services.Configure(this.Configuration.GetSection(“ApplicationConfiguration”));

private ConnectionString connectionString;
///


/// 构造函数 接受配置文件信息
///
public HomeController(IOptions options) {
connectionString = options.Value;

}

public IOptions<ApplicationConfiguration> options;
public HomeController(IUserService _userService, IOptions<ApplicationConfiguration> _appConfiguration) {
    this.userService = _userService;
    this.options = _appConfiguration;
}

public class AppConfigurtaionServices
　　{
　　　　private readonly IOptions _appConfiguration;
　　　　public AppConfigurtaionServices(IOptions appConfiguration)
　　　　{

　　　　}

　　　　public ApplicationConfiguration AppConfigurations
　　　　{
　　　　　　get
